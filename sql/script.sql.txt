DROP TABLE VILLES;
DROP TABLE DEPTS;
DROP TABLE REGIONS;

CREATE TABLE DEPTS (
    id INT AUTO_INCREMENT PRIMARY KEY,
    codeDepartement VARCHAR(255) NOT NULL
);

CREATE TABLE REGIONS (
    id INT AUTO_INCREMENT PRIMARY KEY ,
    nomRegion VARCHAR(255) NOT NULL
);

CREATE TABLE VILLES (
	id INT AUTO_INCREMENT PRIMARY KEY,
	codeRegion VARCHAR(255),
	nomRegion VARCHAR(255),
	codeDepartement VARCHAR(255),
	codeCommune VARCHAR(255),
	nomCommune VARCHAR(255),
	popTotale INT(10),
	idDept INT,
	idRegion INT,
	FOREIGN KEY (idDept) REFERENCES DEPTS(id),
    FOREIGN KEY (idRegion) REFERENCES REGIONS(id)
);
